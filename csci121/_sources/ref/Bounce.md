/*

package j2_bounce_rev3;
import java.awt.Color;
import java.awt.Graphics;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.util.Random;
import javax.swing.JComponent;
import javax.swing.Timer;

/**
 *
 * @author jgoudy
 */
public class Ball extends JComponent {

    //private members
    //Position of the ball
    int xpos = 1;
    int ypos = 1;

    //direction of the ball
    int xdir = 1;
    int ydir = 1;

    //size of the panel: note these are defaults
    int compWidth = 500;
    int compHeight = 500;

    //angle / direction of the balls
    int slopeRun = 1;
    int slopeRise = 1;

    //speed
    int ballspeed = 2;

    //radius of the ball
    int radius = 25;

    //Timer
    Timer bTimer;
    
    //ball color
    Color ballColor;
    
    
     Random RNG;

    //Constructor
    public Ball() {

        xpos = 1;
        ypos = 1;
        
        RNG = new Random();
        
        ballColor = Color.blue;
    }

    //Here is where we draw the ball
    @Override
    public void paintComponent(Graphics g) {

        super.paintComponent(g);
        g.setColor(ballColor);
        g.fillOval(xpos, ypos, radius, radius);

    }


    public void startBall()
    {
        bclock();
    }

    //setters

    //set the compent panel width and height
    public void setCompWidth(int cWidth) {
        compWidth = cWidth - radius;
    }
    public void setCompHeight(int cHeight) {
        compHeight = cHeight - radius;
    }
    //--------------------------------------


    //set the radius
    public void setRadius(int Radius) {
        radius = Radius;
    }
    //--------------------------------------


    //Set the x and y position of the ball
    public void setXpos(int xpos) {
       xpos = xpos;
    }
    public void setYpos(int ypos) {
       ypos = ypos;
    }
    //--------------------------------------


    //set the slope variables
    public void setSlopeRun(int run) {
        slopeRun = run;
    }

    public void setSlopeRise(int rise) {
       slopeRise = rise;
    }
    //--------------------------------------

    //set the speed of the ball
    public void setSpeed(int speed)
    {
        ballspeed = speed;
    }
    
    public void setBallColor(Color ballColor)
    {
        this.ballColor = ballColor;
    }

   
    //create a clock and start it
    //NOte: It is the clock tick that moves the ball
    //NOte: the frame will repaint the screen and show the
    //      location of the ball
    private void bclock() {
        bTimer = new Timer(ballspeed,
                new ActionListener() {

            @Override
            public void actionPerformed(ActionEvent ae) {

                //Check whether the ball is
                //passed the left or right side of the panel
                if (xpos > compWidth) {
                    xdir = xdir * -1;
                    xpos = compWidth - 1;
                } else if (xpos < -1) {
                    xdir = xdir * -1;
                    xpos = 0;
                }


                //Check whether the ball is
                //passed  the top or bottom of the screen
                if (ypos > compHeight) {
                    ydir = ydir * -1;
                    ypos = compHeight;

                } else if (ypos < -1) {
                    ydir = ydir * -1;
                    ypos = 0;

                }

                //increment the x and y position
                xpos = xpos + (xdir * slopeRun);
                ypos = ypos + (ydir * slopeRise);
                
                bTimer.setDelay(RNG.nextInt(100)+ 1);
            }
        }
        );

        //start the timer
        bTimer.start();
    }

}
